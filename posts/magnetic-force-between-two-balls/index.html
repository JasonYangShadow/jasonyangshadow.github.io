Magnetic Force Between Two Balls

https://leetcode.com/problems/magnetic-force-between-two-balls/

class Solution:
    def maxDistance(self, position: List[int], m: int) -> int:
        n = len(position)
        position.sort()
        
        #calculate how many balls can be put into basket
        #what we need is to find proper count(d) == m
        #if count(d) > m, meaning that d is too small otherwise, d is too large.
        def count(d):
            ans, cur = 1, position[0]
            for i in range(1, n):
                if position[i] - cur >= d:
                    ans += 1
                    cur = position[i]
            return ans
        
        l, r = 0, position[-1] - position[0]
        
        #binary search
        while l < r:
            mid = r - (r-l) // 2
            if count(mid) >= m:
                l = mid
            else:
                r = mid - 1
        return l
