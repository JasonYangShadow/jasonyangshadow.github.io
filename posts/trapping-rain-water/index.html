Trapping Rain Water

Given n non-negative integers representing an elevation map where the width of each bar is 1, compute how much water it is able to trap after raining.


The above elevation map is represented by array [0,1,0,2,1,0,1,3,2,1,2,1]. In this case, 6 units of rain water (blue section) are being trapped. Thanks Marcos for contributing this image!

Example:

Input: [0,1,0,2,1,0,1,3,2,1,2,1]
Output: 6

Not good solution

class Solution:
    def trap(self, height: List[int]) -> int:
        l, r = 0, len(height) - 1
        #level indicates the highest threshold, roof
        level = 0
        #low indicates the lowest threshold, floor
        low = 0
        water = 0
        
        while l < r:
            #find minimum one from l,r
            idx = l if height[l] < height[r] else r
            #shrink to middle
            if idx == l:
                l += 1
            else:
                r -= 1
            #curent low value
            low = height[idx]
            #max value equals highest threshold, roof
            level = max(low, level)
            #accumulate water if there is difference between level and low
            water += level - low
        return water

Good solution

class Solution:
    def trap(self, height: List[int]) -> int:
        q = []
        water = 0
        i = 0
        while i < len(height):
            if not q or height[i] < height[q[-1]]:
                q.append(i)
                i += 1
            else:
                p = q.pop()
                if q:
                    mh = min(height[q[-1]], height[i])
                    water += (mh - height[p])*(i - q[-1] -1)
        
        return water
