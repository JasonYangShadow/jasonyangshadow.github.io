3.ABC081B - Shift only

#include <iostream>
#include <vector>
#include <climits>
using namespace std;
int main(){
  int num;
  vector<int> data;
  cin >> num;
  for(int i = 0; i<num;i++){
    int t;
    cin >> t;
    data.push_back(t);
  }

  int minv = INT_MAX;
  for(auto n : data){
    int count = 0;
    while(!(n & 1)){
      count++;
      n = n>>1;
    }
    minv = min(minv, count);
  }
  
  cout<<minv<<endl;
  return 0;
}

5.ABC083B - Some Sums

#include <iostream>
#include <vector>
#include <climits>
using namespace std;
bool valid(int n, int A, int B){
  int val = 0;
  while(n){
    val += n % 10;
    n = n/10;
  }
  return (val>=A && val<=B);
}
int main(){
  int N,A,B;
  cin>>N>>A>>B;
  int ret = 0;
  for(int i = 1; i<=N;i++){
    if(valid(i,A,B)){
      ret += i;
    }
  }
  cout<<ret<<endl;
  return 0;
}

6.ABC088B - Card Game for Two

#include <iostream>
#include <vector>
#include <climits>
#include <algorithm>
using namespace std;

int main(){
  int N;
  cin>>N;
  vector<int> data;
  for(int i = 0; i<N;i++){
  	int val;
    cin>>val;
    data.push_back(val);
  }

  sort(data.begin(), data.end(),greater<int>());
  long alice = 0;
  long bob = 0;
  for(int i = 0; i<N; i++){
    if(i % 2 == 0){
      alice += data[i];
    }else{
      bob += data[i];
    }
  }
  
  cout<<alice - bob<<endl;
  return 0;
}

8.ABC085C - Otoshidama

#include <climits>
#include <algorithm>

int main(){
  int N, Y;
  cin>>N>>Y;
  
  for(int i = 0; i<= N; i++){
    for(int j = 0; j <= N-i; j++){
      int k = N - (i+j);
      if(10000*i + 5000*j + k*1000 == Y){
        cout<<i<<" "<<j<<" "<<k<<endl;
        return 0;
      }
    }
  }
  cout<<"-1 -1 -1"<<endl;

9.ABC049C - 白昼夢 / Daydream

#include <iostream>
#include <algorithm>
#include <stack>
#include <string>
using namespace std;

int main(){
  string S;
  cin>>S;  
  stack<string> st;
  string data[4] = {"dream","dreamer", "erase", "eraser"};
  int skip_len = 0;
  while(S.length()>0){
    int slen = S.length();
    for(int i = 0; i<4; i++){
      if(S.find(data[i]) == 0 && data[i].length()>skip_len){
        st.push(data[i]);
        S = S.substr(data[i].length());
        skip_len = 0;
        break;
      }
    }
    
    if(S.length()== slen){
      if(!st.empty()){
        S = st.top() + S;
        skip_len = st.top().length();
        st.pop();
      }else{
        cout<<"NO"<<endl;
      	return 0;
      }
    }
  }
  cout<<"YES"<<endl;
  return 0;
}

10.ABC086C - Traveling

#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
#include <cmath>
using namespace std;

struct point{
  int t;
  int x;
  int y;
};

bool compare(struct point &p1, struct point &p2){
  int t_diff = abs(p1.t - p2.t);
  int l_diff = abs(p1.x - p2.x) + abs(p1.y - p2.y);
  if(t_diff < l_diff){
    return false;
  }else{
    if(abs(t_diff - l_diff) % 2 == 1){
      return false;
    }
  }
  return true;
}

int main(){
  int N;
  cin>>N;
  vector<struct point> vc;
  for(int i = 0; i<N;i++){
    int t, x, y;
    cin>>t>>x>>y;
    struct point p{t,x,y};
    vc.push_back(p);
  }
  
  struct point start{0,0,0};
  for(auto p : vc){
    if(!compare(start, p)){
      cout<<"No"<<endl;
      return 0;
    }
    start = p;
  }
  cout<<"Yes"<<endl;
  return 0;
}
